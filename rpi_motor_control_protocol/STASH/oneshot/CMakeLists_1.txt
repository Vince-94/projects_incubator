##############
# oneshot125 #
##############

add_executable(oneshot125 oneshot125.c)

pico_set_program_name(oneshot125 "oneshot125")
pico_set_program_version(oneshot125 "0.1")

# Modify the below lines to enable/disable output over UART/USB
pico_enable_stdio_uart(oneshot125 0)
pico_enable_stdio_usb(oneshot125 1)

# Add the standard library to the build
target_link_libraries(oneshot125
  pico_stdlib
  hardware_gpio
  hardware_pwm
  hardware_irq
  hardware_adc
  FreeRTOS-Kernel
  FreeRTOS-Kernel-Heap4
)

# Add the standard include files to the build
target_include_directories(oneshot125 PRIVATE
  ${CMAKE_CURRENT_LIST_DIR}/
  ${CMAKE_CURRENT_LIST_DIR}/..
)

pico_add_extra_outputs(oneshot125)
